from bokeh.palettes import Greys256


source = ColumnDataSource(
    data={
        "x": [x[0] for x in data],
        "y": [x[1] for x in data],
        "name": [x[2] for x in data],
        
        # Accessing the palette with an integer : 255 * grade_between_0_and_1

    ])

glyph = circle.glyph
glyph.line_color = "black"
show(p, notebook_handle=True)
